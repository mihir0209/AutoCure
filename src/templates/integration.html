How It Works

  1. Register your service using the form above (or POST /api/v1/register).
  2. Add the code snippet below to your application. It opens a WebSocket connection to AutoCure and streams your logs.
  3. AutoCure monitors your logs in real-time. When an error is detected, it automatically:
    • Parses the AST of your source code for root-cause analysis
    • Runs AI-powered analysis to identify the root cause
    • Validates confidence across multiple iterations
    • Generates precise fix proposals with exact code diffs
    • Saves an HTML report and optionally emails it to you
  4. View results on the Reports page or check your email.

Python Integration

autocure_client.py — pip install websockets
import asyncio, json, logging, traceback
from datetime import datetime, timezone

try:
    import websockets
except ImportError:
    raise ImportError("Run: pip install websockets")

# Replace with your registered user ID and server address
AUTOCURE_URL = "{{ ws_url }}/ws/logs/YOUR_USER_ID"


class AutoCureHandler(logging.Handler):
    """Python logging handler that streams logs to AutoCure via WebSocket."""

    def __init__(self, ws_url=AUTOCURE_URL, level=logging.WARNING):
        super().__init__(level)
        self.ws_url = ws_url
        self._queue = asyncio.Queue()
        self._task = None

    def start(self):
        """Call this after your event loop is running."""
        self._task = asyncio.ensure_future(self._sender())

    async def _sender(self):
        """Maintain a persistent WebSocket connection and stream log entries."""
        while True:
            try:
                async with websockets.connect(self.ws_url) as ws:
                    print(f"[AutoCure] Connected to {self.ws_url}")
                    while True:
                        entry = await self._queue.get()
                        await ws.send(json.dumps(entry))
            except Exception as e:
                print(f"[AutoCure] Connection lost ({e}), reconnecting in 5s...")
                await asyncio.sleep(5)

    def emit(self, record):
        entry = {
            "message": record.getMessage(),
            "level": record.levelname,
            "timestamp": datetime.now(timezone.utc).isoformat(),
            "source_file": record.pathname,
            "line_number": record.lineno,
            "logger_name": record.name,
        }
        if record.exc_info and record.exc_info[0]:
            entry["stack_trace"] = "".join(
                traceback.format_exception(*record.exc_info)
            )
        try:
            self._queue.put_nowait(entry)
        except Exception:
            pass

Node.js Integration

autocure-client.js — npm install ws
const WebSocket = require('ws');

// Replace with your registered user ID and server address
const AUTOCURE_URL = '{{ ws_url }}/ws/logs/YOUR_USER_ID';

class AutoCureClient {
  constructor(url) {
    this.url = url || AUTOCURE_URL;
    this.ws = null;
    this.queue = [];
    this.connect();
  }

  connect() {
    this.ws = new WebSocket(this.url);
    this.ws.on('open', () => {
      console.log('[AutoCure] Connected to', this.url);
      // Flush queued messages
      this.queue.forEach(e => this.ws.send(JSON.stringify(e)));
      this.queue = [];
    });
    this.ws.on('close', () => {
      console.log('[AutoCure] Disconnected, reconnecting in 5s...');
      setTimeout(() => this.connect(), 5000);
    });
    this.ws.on('message', (raw) => {
      try {
        const msg = JSON.parse(raw);
        if (msg.type === 'analysis_complete') {
          console.log('[AutoCure] Analysis result:', msg.payload);
        }
      } catch (e) { /* ignore */ }
    });
    this.ws.on('error', () => {}); // suppress unhandled
  }

  send(entry) {
    const payload = {
      message: entry.message || '',
      level: entry.level || 'INFO',
      timestamp: new Date().toISOString(),
      source_file: entry.file || '',
      line_number: entry.line || 0,
      stack_trace: entry.stack || '',
      api_endpoint: entry.endpoint || '',
      http_method: entry.method || '',
    };
    if (this.ws && this.ws.readyState === WebSocket.OPEN) {
      this.ws.send(JSON.stringify(payload));
    } else {
      this.queue.push(payload);
    }
  }

  error(message, err, extra) {
    this.send({
      message: message,
      level: 'ERROR',
      stack: err ? err.stack : '',
      ...(extra || {}),
    });
  }
}
